CREATE PROC [dbo].[cafAddEmpProject]
AS
DECLARE
@counter int,
@ProjectID int,
@EmployeeID int
SET @counter = 0
WHILE @counter < 100
BEGIN
	SET @ProjectID = (SELECT TOP 1 P.ProjectID FROM Project P ORDER BY NEWID())
	SET @EmployeeID = (SELECT TOP 1 E.EmployeeID FROM Employee E ORDER BY NEWID())
	SET @counter = @counter + 1
	INSERT INTO Employee_Project(EmployeeID, ProjectID)
		VALUES(@EmployeeID, @ProjectID)
END
